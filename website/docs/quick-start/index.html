  return (
    <main>
      <h1>Welcome to Jen.js</h1>
      <p>This is your home page!</p>
    </main>

Data Loading

Load data from your loader function:

import type { LoaderContext } from "@jenjs/master/src/core/types.js";

export async function loader(ctx: LoaderContext) {
  const posts = await fetchBlogPosts();
  return { posts };
}

export default function BlogPage({ data }: { data: { posts: Post[] } }) {
  return (
    <main>
      <h1>Blog</h1>
      <ul>
        {data.posts.map(post => (
          <li key={post.id}>{post.title}</li>
        ))}
      </ul>
    </main>
  );
}

Interactive Components (Islands)

Add interactivity with islands:

import { useState } from 'preact/hooks';
import { Island } from "@jenjs/master/src/runtime/islands.js";

function CounterImpl() {
  const [count, setCount] = useState(0);
  return (
    <button onClick={() => setCount(c => c + 1)}>
      Count: {count}
    </button>
  );
}

const Counter = Island(CounterImpl, "load");

export default function HomePage() {
  return (
    <main>
      <h1>Counter</h1>
      <Counter /> {/* This component is interactive */}
    </main>
  );
}

API Routes

Create an API endpoint site/api/(hello).ts:

import type { ApiHandler } from "@jenjs/master/src/server/api-routes.js";

export const GET: ApiHandler = async (ctx) => {
  return { message: "Hello World!" };
};

export const POST: ApiHandler = async (ctx) => {
  const { name } = ctx.body;
  return { message: `Hello, ${name}!` };
};

Access it at http://localhost:3000/api/hello.

Rendering Modes

Control how each page is rendered:

// Static generation (default)
export const mode = "ssg";

// Server-side rendering
export const mode = "ssr";

// Incremental Static Regeneration
export const mode = "isr";
export const revalidateSeconds = 60;

export default function Page() {
  return <h1>Rendered with custom mode</h1>;
}

Build and Deploy

Build your site for production:

npm run build

This generates a static site in dist/. Deploy to GitHub Pages, Vercel, Netlify, or any static host.

Start the Dev Server

npm run dev
