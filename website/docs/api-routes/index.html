API Routes

API routes let you build backend functionality alongside your front-end pages.

Basic API Route

Create a file site/api/(users).ts:

import type { ApiHandler } from "@jenjs/master/src/server/api-routes.js";

export const GET: ApiHandler = async (ctx) => {
  const users = await db.query({ find: 'users' });
  return { users };
};

export const POST: ApiHandler = async (ctx) => {
  const { name, email } = ctx.body;
  const user = await db.create('users', { name, email });
  return { user };
};

Access at GET /api/users and POST /api/users.

Request Context

Access request details via the ctx parameter:

export const POST: ApiHandler = async (ctx) => {
  const { req, res, url, method, query, params, body, headers, cookies } = ctx;
  
  // URL and routing
  console.log(url.pathname);
  console.log(method);
  console.log(params);        // Route parameters
  console.log(query);         // Query string
  
  // Request data
  console.log(body);          // Parsed JSON body
  console.log(headers);       // All headers
  console.log(cookies);       // Parsed cookies
  
  // Low-level access
  console.log(req);           // Node.js IncomingMessage
  console.log(res);           // Node.js ServerResponse
  
  return { ok: true };
};

Dynamic Routes

Create dynamic API routes with parameters:

// site/api/users/([id]).ts
export const GET: ApiHandler = async (ctx) => {
  const { id } = ctx.params;
  const user = await db.query({ find: 'users', where: { id } });
  return { user };
};

export const PUT: ApiHandler = async (ctx) => {
  const { id } = ctx.params;
  const { name } = ctx.body;
  await db.update('users', { id }, { name });
  return { ok: true };
};

export const DELETE: ApiHandler = async (ctx) => {
  const { id } = ctx.params;
  await db.delete('users', { id });
  return { ok: true };
};

Access at /api/users/123.

Response Types

Return different response types:

// JSON object (auto-serialized)
export const GET: ApiHandler = async (ctx) => {
  return { message: 'Hello' };
};

// Raw Response
export const GET: ApiHandler = async (ctx) => {
  return new Response('Plain text', {
    status: 200,
    headers: { 'Content-Type': 'text/plain' }
  });
};

// JSON with status code
export const POST: ApiHandler = async (ctx) => {
  return new Response(JSON.stringify({ id: 1 }), {
    status: 201,
    headers: { 'Content-Type': 'application/json' }
  });
};

// Null to skip response
export const GET: ApiHandler = async (ctx) => {
  ctx.res.writeHead(200, { 'Content-Type': 'application/json' });
  ctx.res.end(JSON.stringify({ data: [] }));
  return null;
};

Query Parameters

Access query string parameters:

// GET /api/search?q=javascript&limit=10
export const GET: ApiHandler = async (ctx) => {
  const { q, limit } = ctx.query;
  const results = await search(q, parseInt(limit || '20'));
  return { results };
};

Request Headers

Read and set headers:

export const POST: ApiHandler = async (ctx) => {
  // Read headers
  const contentType = ctx.headers['content-type'];
  const authorization = ctx.headers['authorization'];
  
  // Set response headers
  ctx.res.setHeader('Cache-Control', 'max-age=3600');
  ctx.res.setHeader('X-Custom-Header', 'value');
  
  return { ok: true };
};

Cookies

Read and set cookies:

export const POST: ApiHandler = async (ctx) => {
  // Read cookies
  const sessionId = ctx.cookies.sessionId;
  
  // Set cookies (via headers)
  ctx.res.setHeader('Set-Cookie', 'sessionId=abc123; Max-Age=3600');
  
  return { ok: true };
};

Error Handling

Handle errors gracefully:

export const GET: ApiHandler = async (ctx) => {
  try {
    const data = await fetchData();
    return { data };
  } catch (error) {
    return new Response(JSON.stringify({ error: 'Failed to fetch' }), {
      status: 500,
      headers: { 'Content-Type': 'application/json' }
    });
  }
};

Authentication Example

import { verifyToken } from '@src/auth/jwt.js';

export const GET: ApiHandler = async (ctx) => {
  const auth = ctx.headers['authorization'];
  if (!auth) {
    return new Response(JSON.stringify({ error: 'Unauthorized' }), {
      status: 401,
      headers: { 'Content-Type': 'application/json' }
    });
  }
  
  const token = auth.replace('Bearer ', '');
  const payload = verifyToken(token);
  
  return { user: payload };
};

CORS Headers

Enable CORS:

export const GET: ApiHandler = async (ctx) => {
  ctx.res.setHeader('Access-Control-Allow-Origin', '*');
  ctx.res.setHeader('Access-Control-Allow-Methods', 'GET, POST, PUT, DELETE');
  ctx.res.setHeader('Access-Control-Allow-Headers', 'Content-Type');
  
  return { ok: true };
};
