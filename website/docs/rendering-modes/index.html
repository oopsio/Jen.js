Rendering Modes

Jen.js supports four rendering modes. Choose the best one for each route.

Static Site Generation (SSG)

Default mode. Pages are pre-rendered at build time.

export const mode = "ssg";

export async function loader(ctx: LoaderContext) {
  // This runs at build time
  const posts = await fetchPosts();
  return { posts };
}

export default function BlogPage({ data }: { data: { posts: Post[] } }) {
  return (
    <main>
      <h1>Blog</h1>
      {data.posts.map(post => <article key={post.id}>{post.title}</article>)}
    </main>
  );
}

Server-Side Rendering (SSR)

Pages render on each request.

export const mode = "ssr";

export async function loader(ctx: LoaderContext) {
  // This runs on every request
  const user = await getUser(ctx.cookies.userId);
  return { user };
}

export default function DashboardPage({ data }: { data: { user: User } }) {
  return (
    <main>
      <h1>Welcome, {data.user.name}</h1>
    </main>
  );
}

Incremental Static Regeneration (ISR)

Static pages that revalidate on a schedule.

export const mode = "isr";
export const revalidateSeconds = 3600; // Revalidate every hour

export async function loader(ctx: LoaderContext) {
  // Runs at build time, then again after revalidateSeconds
  const products = await fetchProducts();
  return { products };
}

export default function ProductsPage({ data }: { data: { products: Product[] } }) {
  return (
    <main>
      <h1>Products</h1>
      {data.products.map(p => <div key={p.id}>{p.name}</div>)}
    </main>
  );
}

Partial Pre-Rendering (PPR)

Static shell with dynamic content zones.

export const mode = "ppr";

export async function loader(ctx: LoaderContext) {
  // Returns both static and dynamic data
  return {
    static: {
      title: "Dashboard"
    },
    dynamic: {
      userStats: null // Placeholder, fetched on client
    }
  };
}

export default function DashboardPage({ data }: { data: any }) {
  return (
    <main>
      <h1>{data.static.title}</h1>
      {/* Static content renders at build time */}
      <StaticContent />
      
      {/* Dynamic content fetched on request */}
      {data.dynamic?.userStats && <DynamicStats stats={data.dynamic.userStats} />}
    </main>
  );
}
