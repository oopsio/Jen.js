Configuration

jen.config.ts

The jen.config.ts file in your project root configures the Jen.js framework.

import type { FrameworkConfig } from "@jenjs/master/src/core/config.js";

const config: FrameworkConfig = {
  siteDir: "site",           // Routes directory
  distDir: "dist",           // Build output
  
  routes: {
    fileExtensions: [".tsx", ".jsx", ".ts", ".js"],
    routeFilePattern: /^\((.+)\)\.(t|j)sx?$/,
    enableIndexFallback: true
  },
  
  rendering: {
    defaultMode: "ssg",
    defaultRevalidateSeconds: 0
  },
  
  css: {
    globalScss: "site/styles/global.scss",
    criticalBudget: 4096,
    extractCritical: true
  },
  
  build: {
    minifyHtml: true,
    minifyCss: true,
    minifyJs: true,
    hashAssets: true,
    generateSitemap: true,
    incrementalBuild: true
  },
  
  server: {
    port: 3000,
    hostname: "0.0.0.0"
  }
};

export default config;

Configuration Options

siteDir

Directory containing your routes.

siteDir: "site"

distDir

Output directory for built site.

distDir: "dist"

routes

Route configuration.

routes: {
  // File extensions to look for
  fileExtensions: [".tsx", ".jsx", ".ts", ".js"],
  
  // Pattern for route files
  routeFilePattern: /^\((.+)\)\.(t|j)sx?$/,
  
  // Enable index fallback for directories
  enableIndexFallback: true
}

rendering

Default rendering mode and ISR revalidation.

rendering: {
  defaultMode: "ssg",      // "ssg" | "ssr" | "isr" | "ppr"
  defaultRevalidateSeconds: 0
}

css

SCSS and CSS configuration.

css: {
  // Global SCSS file
  globalScss: "site/styles/global.scss",
  
  // Critical CSS size budget
  criticalBudget: 4096,
  
  // Extract critical CSS
  extractCritical: true
}

build

Build optimization options.

build: {
  minifyHtml: true,      // Minify HTML output
  minifyCss: true,       // Minify CSS
  minifyJs: true,        // Minify JavaScript
  hashAssets: true,      // Hash asset filenames
  generateSitemap: true, // Generate sitemap.xml
  incrementalBuild: true // Cache builds
}

server

Development server configuration.

server: {
  port: 3000,
  hostname: "0.0.0.0"
}

Environment Variables

Create a .env file for environment-specific configuration:

# Database
DATABASE_URL=sqlite://./data.db

# API Keys
API_KEY=your-key-here

# Build
GENERATE_SITEMAP=true
MINIFY=true

Access in your routes:

export async function loader(ctx: LoaderContext) {
  const dbUrl = process.env.DATABASE_URL;
  return { };
}

Conditional Configuration

Adjust config based on environment:

const isProduction = process.env.NODE_ENV === 'production';

const config: FrameworkConfig = {
  build: {
    minifyHtml: isProduction,
    minifyCss: isProduction,
    minifyJs: isProduction,
  },
  // ...
};

export default config;
