API Reference

Core APIs

LoaderContext

Data loading context available in route loaders.

interface LoaderContext {
  url: URL;                    // Full request URL
  params: Record<string, string>; // Route parameters
  query: Record<string, string>;  // Query string
  headers: Record<string, string>; // Request headers
  cookies: Record<string, string>; // Parsed cookies
  data?: Record<string, any>;  // Data from middleware
}

RouteModule

Exports available in route files.

type RouteModule = {
  mode?: "ssg" | "ssr" | "isr" | "ppr";
  revalidateSeconds?: number;
  hydrate?: boolean;
  middleware?: RouteMiddleware | RouteMiddleware[];
  loader?: (ctx: LoaderContext) => Promise<any> | any;
  Head?: (props: any) => any;
  default: (props: any) => any;
};

Island Function

Mark a component as an island.

function Island<T>(
  component: ComponentType<T>,
  strategy: "load" | "idle" | "visible"
): ComponentType<T>

API Routes

ApiRouteContext

Context for API route handlers.

interface ApiRouteContext {
  req: IncomingMessage;
  res: ServerResponse;
  url: URL;
  method: string;
  query: Record<string, string>;
  params: Record<string, string>;
  headers: Record<string, string>;
  cookies: Record<string, string>;
  body: any;
}

HTTP Methods

Define handlers for specific HTTP methods:

export const GET: ApiHandler = async (ctx) => {
  return { data: [] };
};

export const POST: ApiHandler = async (ctx) => {
  return new Response(JSON.stringify({ id: 1 }), {
    status: 201,
    headers: { "Content-Type": "application/json" }
  });
};

export const PUT: ApiHandler = async (ctx) => {
  return { updated: true };
};

export const DELETE: ApiHandler = async (ctx) => {
  return { deleted: true };
};

Database

createDB

Create a database connection.

import { createDB } from "@jenjs/master/src/db/index.js";

const db = createDB({
  type: 'sqlite' | 'mysql' | 'postgres' | 'jdb',
  connection: './data.db'
});

await db.connect();

Query Methods

// Find documents
const users = await db.query({ find: 'users', where: { active: true } });

// Create
const user = await db.create('users', { name: 'Alice' });

// Update
await db.update('users', { id: 1 }, { name: 'Bob' });

// Delete
await db.delete('users', { id: 1 });

// Count
const count = await db.count('users', { active: true });

Caching

MemoryCache

In-memory caching.

import { MemoryCache } from "@jenjs/master/src/cache/memory.js";

const cache = new MemoryCache();
cache.set('key', 'value', 3600);
const value = cache.get('key');

RedisCache

Redis-backed caching.

import { RedisCache } from "@jenjs/master/src/cache/redis.js";

const cache = new RedisCache({ host: 'localhost', port: 6379 });
await cache.connect();
await cache.set('key', 'value', 3600);

Authentication

SessionManager

Session management.

import { SessionManager } from "@jenjs/master/src/auth/session.js";

const sessions = new SessionManager();
sessions.createSession(userId, { role: 'admin' });
const session = sessions.getSession(userId);

Internationalization

I18n

Translation management.

import { I18n } from "@jenjs/master/src/i18n/index.js";

const i18n = new I18n('en');
i18n.setLocale('es');
const text = i18n.t('greeting');
